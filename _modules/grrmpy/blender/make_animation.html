[ドキュメント]def make_animation(folder,traj,step=3,start=0,color_dict=True,
                   pyfile="animation.py",x3dfile="Atoms.x3d",pklfile="property.pkl"):
    """
    
    AtomsのImagesからBlenderアニメーションを作成するためのPythonスクリプトを作成する

    Parameters:
    
    folder: str
        | Pythonスクリプトなどの必要なファイルを格納するためのフォルダ
    traj: Trajectory or list of Atoms
        | TrajectoryまたはAtomsのリスト
        | 構造の数が多い場合はTrajectoryで作成しないとメモリが溢れてしまう.
    step: int
        | stepフレーム毎にキーを打つ
    start: int
        startフレームからキーを打つ
    color_dict: bool or dict
        | Trueにした場合元素毎に着色される
        | 色を指定したい場合は辞書型で与える(元素:(R,G,B,A)) RGBは1で規格化する必要がある.
        | ex) {'Si':(1.0, 1.0, 0.0, 1),'O':(1.0, 0.0, 0.0, 1),`Al`:(0.235, 1.0, 0.0, 1)}
        | VESTAのelement.iniからcolor_dictを作成する場合は
          :class:`grrmpy.io.read_elementsini.read_elementsini` を用いると良い 
    """
    p = Path(folder)
    if p.exists() and len(list(p.iterdir()))!=0:
        raise Exception(f"{p.name}内に既にファイルが存在します")
    else:
        p.mkdir()
    make_py_script(p/pyfile,p/x3dfile,p/pklfile,step=3,start=0)
    write(p/x3dfile,traj[0],format="x3d")
    prop2pkl(p/pklfile,traj,color_dict)

bpy.ops.import_scene.x3d(filepath=x3d_path)

frame_num = start

with open(pkl_path,'rb') as f:
    nimages = pickle.load(f)
    chemical_symbols = pickle.load(f)
    color_dict = pickle.load(f)
    
    if color_dict:
        mat_dict = {}
        for symbol in set(color_dict.keys()):
            mat = bpy.data.materials.new(symbol)
            mat.use_nodes = True
            bsdf = mat.node_tree.nodes["Principled BSDF"]
            bsdf.inputs[0].default_value = color_dict[symbol]
            mat_dict[symbol] = mat
    
    for obj,symbol in zip(bpy.data.objects[-nimages:],chemical_symbols):
        if color_dict:
            obj.data.materials.append(mat_dict[symbol])
        obj.modifiers.new("subd", type='SUBSURF')
        obj.modifiers['subd'].levels = 3
        obj.modifiers['subd'].render_levels = 3
    
    while 1:
        try:
            bpy.context.scene.frame_set(frame_num)
            positions = pickle.load(f)
            for obj,position in zip(bpy.data.objects[-nimages:],positions):
                obj.location = position.tolist()
                obj.keyframe_insert(data_path = 'location',index = -1)
            frame_num += step
        except EOFError:
            break
"""

def make_py_script(file,x3dfile,pklfile,step=3,start=0):
    x3d = Path(x3dfile)
    pkl = Path(pklfile)
    with open(file,"w") as f:
        f.write(text1)
        f.write(f"x3d_path = str(p.parent.with_name('{x3d.name}').resolve())\n")
        f.write(f"pkl_path = str(p.parent.with_name('{pkl.name}').resolve())\n")
        f.write(f"step = {step}\nstart= {start}")
        f.write(text2)
