def _make_ball_and_stick_pyscript(atoms,**kwargs):
    structure = Structure(atoms)
    bond_dict = kwargs.get("bonds",{"mult":1})
    mult = bond_dict.pop("mult",1)
    bond_kwargs = bond_dict
    data = {
        "atoms":atoms,
        "size_mult":kwargs.get("size_mult",default_size_mult),
        "colors":structure.extract_colors(kwargs.get("colors",{})),
        "ball_sizes":structure.extract_sizes(kwargs.get("ball_sizes",{})),
        "bond_color":kwargs.get("bond_color",default_bond_color),
        "bond_radius":kwargs.get("bond_radius",default_bond_radius),
        "positions":atoms.get_positions().tolist(),
        "bonds":structure.get_bond(mult=mult,**bond_kwargs)
    }
    return 'ball_and_stick.py', data

def _make_animation(folder,traj,**kwargs):
    p = Path(__file__).parent
    folder = Path(folder)
    if not folder.suffix == ".zip":
        raise Exception("拡張子はzipである必要があります")
    structure = Structure(traj[0])
    pyfile = kwargs.get("pyfile",default_pyfile)
    pklfile = kwargs.get("pklfile",default_pklfile)
    data = {
        "step":kwargs.get("step",default_step),
        "ball_sizes":structure.extract_sizes(kwargs.get("ball_sizes",{})),
        "start":kwargs.get("start",default_start),
        "colors":structure.extract_colors(kwargs.get("colors",{})),
        "pklfile":Path(pklfile)
    }
    # pickleの中身["nimages","elements","all_positions"]
    env = Environment(loader=FileSystemLoader(p/"templates/", encoding='utf8'))
    template = env.get_template("animation.py")
    script = template.render(data)
    with zipfile.ZipFile(folder, 'w') as zf:
        with zf.open(pyfile, 'w') as f:
            f.write(script.encode('utf-8'))
        with zf.open(pklfile,"w") as f:
            pickle.dump(len(traj), f)
            pickle.dump(traj[0].get_chemical_symbols(), f)
            for atoms in traj:
                pickle.dump(atoms.get_positions().tolist(), f)


[ドキュメント]def make_py_script(file,atoms,style="ball_and_stick",**kwargs):
    """Blenderに構造をインポートするためのPythonスクリプトを作成する

    Parameters:
    
    file: str, path object
        | アニメーションを作成する場合,zipファイル
        | 単一構造の場合,pyファイル
    atoms: Atoms, list of Atoms, Trajectory
        | アニメーションを作成する場合はTrajectoryまたはAtomsのリスト
        | 単一構造の場合,Atoms
    style: str
        | 'ball_and_stick','space_filling','stick',のいずれか
        | 引数のatomsがリストまたはTrajectoryの場合(アニメーションを作成する場合),設定する必要はない.
    kwargs:
        | styleによってとることのできるキーワード引数は違う.
        | ドキュメントを参照する.
    """
    if not type(atoms) == Atoms:
        _make_animation(file, atoms,**kwargs)
        return
    if style == "ball_and_stick":
        if kwargs.get("bicolor",bicolor):
            file_name,data = _make_bicolor_pyscript(atoms,**kwargs)
        else:
            file_name,data = _make_ball_and_stick_pyscript(atoms,**kwargs)
    elif style == "space_filling":
        file_name,data = _make_space_filling_pyscript(atoms,**kwargs)
    elif style == "stick":
        file_name,data = _make_stick_pyscript(atoms,**kwargs)
    p = Path(__file__).parent
    env = Environment(loader=FileSystemLoader(p/"templates/", encoding='utf8'))
    template = env.get_template(file_name)
    script = template.render(data)
    with open(file,"w") as f:
        f.write(script)
