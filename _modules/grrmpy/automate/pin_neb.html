class PinNebExecutor(Executor):
    """
    
    | NEBダイアグラム中に中間体とみられるエネルギーの谷間があれば,最適化計算する.
    | それが新たな中間体と判断した際はNEBイメージを切断し(つまり素反応に切り分けて)それぞれをNEB計算する.
    | 中間体らしき構造を見つける度にこれを行ない,全ての素反応のNEB計算を行なう.

    Parameters:
    
    images: list of images
        | 初期images. calculatorを設定すう必要はない
    constraints: ASE constraint object
        | ASEのConstraint
    parallel: bool
        | NEBを並列計算する場合True
    mic: bool
        | 最小イメージ規則を適用する場合Ture.Noneの場合与えられたimagesが周期境界の場合Trueにする.
    calc_func: function object
        | calculatorを返す関数
        
    Example:
    
        >>> from grrmpy.automate import Queue,PinNEB,is_similar
        >>> pin_neb = PinNEB(images)
        >>> que = Queue(pin_neb)
        >>> que.run()
    """

    def make_path_txt(self,path_txt,ini_name,fin_name,add_data):
        """
        path_txt: 現在のpath_txt
        add_data: list of int
            EQを追加する(EQのindex番号のリスト)
        add_data: int:
            TSを追加する
        add_data: None
            バリアレスを追加
        """
        if add_data is None:
            new_path_txt = re.sub(f"(EQ{ini_name}).(EQ{fin_name})",f"\\1~\\2",path_txt)
        elif type(add_data) == int:
            new_path_txt = re.sub(f"(EQ{ini_name}).(EQ{fin_name})",f"\\1;TS{add_data};\\2",path_txt)
        elif type(add_data)==str:
            new_path_txt = re.sub(f"EQ{ini_name}.EQ{fin_name}",f"{add_data}",path_txt)
        return new_path_txt

    def set_param(self,**param):
        """パラメータを設定する
        
        現在適応されているパラメータは `get_param()`

        Example:
        
            >>> pin_neb.set_param(minenergy=10)
        """
        default_param = self.param
        unexpected_argument = set(param)-set(default_param)
        if len(unexpected_argument) != 0:
            raise Exception(f"予期しない引数({', '.join(unexpected_argument)})が存在します.\n"+
                            f"'self.default_param??'で引数を確認できます\n"
                           f"Parameters:\n{', '.join(self.default_param.keys())}")
        self.param.update(param)
        for key,val in default_param.items():
            setattr(self, key, val)

    def run_and_push(self,data,layer)->list:
        """必須メソッド"""
        ea,(self.ini_name,self.fin_name,self.num),images = data
        pre_file_name = f"pre_EQ{self.ini_name}-EQ{self.fin_name}({layer}-{self.num})"
        file_name = f"EQ{self.ini_name}-EQ{self.fin_name}({layer}-{self.num})"
        try:
            ####PreNEB####
            _,images = self.run_neb(images,name=pre_file_name,preneb=True) # PreNEB
            # 極小値を探す
            minimum_point,_ = find_local_minimum_points_and_ea(images,minenergy=self.minenergy)
            write_html(f"{pre_file_name}.html",images,highlight=minimum_point)
            if len(minimum_point) > 0:
                push_data = self.run_opt_mkimg(images,minimum_point,pre_file_name)
                if push_data is not None:
                    return push_data
            #収束させる(NEB)
            converged,images = self.run_neb(images,name=file_name,preneb=False) # NEB
            # 極小値を探す
            minimum_point,_ = find_local_minimum_points_and_ea(images,minenergy=self.minenergy) 
            if converged:
                if len(minimum_point)==0 or not self.separate_completely:
                    write(f"{self.img_folder}/NEB{layer}-{self.num}(EQ{self.ini_name}-TS{len(self.ts_list)}-EQ{self.fin_name}).traj",self.neb.images)
                    imax = get_imax(images,self.barrier_less,self.calc_func)
                    if imax is not None:
                        self.register_ts(images[imax].copy())
                        self.write_path_dat_and_html(self.ini_name,self.fin_name,len(self.ts_list)-1)
                    else:
                        # バリアレスの場合
                        self.write_path_dat_and_html(self.ini_name,self.fin_name,None)
                else:
                    push_data = self.run_opt_mkimg(images,minimum_point,file_name)
                    if push_data is not None:
                        return push_data
                    else:
                        write(f"{self.img_folder}/NEB{layer}-{self.num}(EQ{self.ini_name}-TS{len(self.ts_list)}-EQ{self.fin_name}).traj",self.neb.images)
                        imax = get_imax(images,self.barrier_less,self.calc_func)
                        if imax is not None:
                            self.register_ts(images[imax].copy())
                            self.write_path_dat_and_html(self.ini_name,self.fin_name,len(self.ts_list)-1)  
            else:
                if len(minimum_point)==0:
                    write(f"{self.unconv_img_folder}/NEB{layer}-{self.num}(EQ{self.ini_name}-EQ{self.fin_name}).traj",self.neb.images)
                else:
                    push_data = self.run_opt_mkimg(images,minimum_point,file_name)
                    if push_data is not None:
                        return push_data
                    else:
                        write(f"{self.unconv_img_folder}/NEB{layer}-{self.num}(EQ{self.ini_name}-EQ{self.fin_name}).traj",self.neb.images)   
            return []
        except Exception as e:
            self.write_errorlog(f"EQ{self.ini_name}-EQ{self.fin_name}の計算:\n{e}\n{traceback.format_exc()}\n")
            return []

    def run_neb(self,images,name,preneb=False):
        self.neb = RepetitiveNEB(images,
                                 constraints=self.constraints,
                                 parallel=self.parallel,
                                 mic=self.mic,
                                 calc_func=self.calc_func,
                                 name=name)
        self.neb.attach(lambda:write_html(f"{name}.html",self.neb.images),interval=10)
        if preneb:
            self.neb.attach(lambda:write(f"{name}.traj",self.neb.images),interval=10)
            param = self.preneb_param
        else:
            self.neb.attach(lambda:write(f"Current.traj",self.neb.images),interval=10)
            param = self.neb_param
            rename_list = ["log","html"]
            for r in rename_list:
                if Path(f"pre_{name}.{r}").exists():
                    Path(f"pre_{name}.{r}").rename(f"{name}.{r}")
            if Path(f"pre_{name}.traj").exists():
                Path(f"pre_{name}.traj").unlink()      
        converged = self.neb.run(**param)
        return converged, self.neb.images
    
    def run_opt(self,atoms,name):
        # atomsにはconstraintsがかかっているハズ
        self.opt = RepetitiveOpt(atoms,constraints=self.constraints,logfile=f"{name}.log",calc_func=self.calc_func)
        # self.opt.attach(lambda:write(f"{name}.traj",self.opt.atoms))
        converged = self.opt.run(**self.opt_param)
        return converged,self.opt.atoms
    
    def run_opt_mkimg(self,images,minimum_point,name):
        # 最適化-->ジオメトリの検証-->images作成-->queueに渡す
        eq_result_list = []
        eq_num = []
        for min_idx in minimum_point:
            conv,atoms = self.run_opt(images[min_idx].copy(),f"opt_{name}-{min_idx}")
            if conv:
                self.register_eq(atoms)
                eq_result_list.append([conv,atoms]) #収束状況,Atoms,EQ番号
                eq_num.append(len(self.eq_list)-1)
            else:
                eq_result_list.append(conv,atoms)
                
        write_html(f"{name}.html",images,highlight=minimum_point,
                    annotation=[(i,f"Succeed-->EQ{eq_idx}" if conv else "Failed") 
                                for  i,(conv,_),eq_idx in zip(minimum_point,eq_result_list,eq_num)])
        # self.write_path_dat_and_html(self.ini_name,self.fin_name,eq_num)
        eq_atoms = [images[0]]+[atoms for conv,atoms in eq_result_list if conv]+[images[-1]] #収束したもの+元々のini,finのみ抽出
        eq_num_list = [self.ini_name] + eq_num + [self.fin_name]
        all_ini_fin_list = [[eq_atoms[i],eq_atoms[i+1],eq_num_list[i],eq_num_list[i+1]] for i in range(len(eq_atoms)-1)] # [[iniAtoms,finAtoms],]
        
        #ジオメトリの検証
        ini_fin_list = []
        sign = [] # |異なる構造 ;同じ構造
        for ini,fin,ini_eq,fin_eq in all_ini_fin_list:
            if not self.isomorphism_func(ini,fin):
                ini_fin_list.append([ini,fin,ini_eq,fin_eq])
                sign.append("|")
            else:
                sign.append(";")
        path_txt = f"EQ{eq_num_list[0]}"
        for s,ini_idx in zip(sign,eq_num_list[1:]):
            path_txt+=f"{s}EQ{ini_idx}"
        self.write_path_dat_and_html(self.ini_name,self.fin_name,path_txt)
        
        
        # images作成
        push_data = []
        if len(ini_fin_list) != 0:
            for i,(ini,fin,ini_eq,fin_eq) in enumerate(ini_fin_list):
                if type(self.nimages)==int:
                    n = self.nimages
                else:
                    n = get_appropriate_nimages(ini,fin,dist=self.nimages[0],nmax=self.nimages[1],nmin=self.nimages[2],mic=self.mic)
                images = [ini.copy() for _ in range(n)]+[fin.copy()]
                interpolate(images,mic=self.mic)
                set_calc2images(images,self.calc_func)
                set_const2images(images, self.constraints)
                ea = get_ea(images,calc_func=self.calc_func)
                push_data.append([ea,[ini_eq,fin_eq,i],images])
            return push_data
        return None
    
    def data_to_serializable_obj(self,data):

[ドキュメント]def is_similar(atoms1,atoms2,energy_dif=10,bond_dif=(1,1.0),mult=1,**kwargs):
    """atoms1とatoms2を同一構造とみなすか(NEBでつなぐ必要はないか?)
    
    PinNEBの引数:isomorphism_funcのための関数
    
    基本的には結合状態が同じ且つ,エネルギー差がenergy_dif(kJ/mol)以下であれば同一構造とみなす.
    また,bond_dif[0] 本以下の結合状態の違いであり,その結合の差がbond_dif[1] Å以下の時も同一構造とする.
    
    Parameters:
    
    atoms1: Atoms
        Atomsオブジェクト
    atoms2: Atoms:
        Atomsオブジェクト
    energy_dif: float
        同じ結合状態かつ,エネルギー差がenergy_dif(kJ/mol)の時,同一構造とする
    bond_dif: Tuple[int,flaot]
        | 1要素目は許容できる結合の差(結合本数)
        | 2要素目は結合状態の異なる部分の許容できる結合距離の差(Å)
    """
    energy1 = atoms1.get_potential_energy()
    energy2 = atoms2.get_potential_energy()
    if abs(energy1-energy2)*mol/kJ > energy_dif:
        # energy_dif以下のエネルギーであるか
        return False
    formation, cleavage = get_bond_diff(atoms1,atoms2,mult=1,**kwargs)
    if len((dif_idx:=formation+cleavage))>bond_dif[0]:
        # bond_dif[0]以下の結合状態の違いでしかないか
        return False
    for a_idx,b_idx in dif_idx:

        d2 = atoms2.get_distance(a_idx,b_idx)
        if abs(d1-d2)>bond_dif[1]:
            return False
